CXX = g++
CXXFLAGS = \
	-I"D:\Box2D" \
	-I"Framework"
LINKFLAGS = \
	-L"D:\Box2D\Build\Box2D" \
	-L"D:\Box2D\Build\freeglut" \
	-L"D:\Box2D\Build\glui" \
	-lBox2D \
	-lfreeglut_static \
	-lglui -lopengl32 -lglu32 \
	-lwinmm -lgdi32

OBJS = \
	obj\Main.o \
	obj\Render.o \
	obj\Test.o \
	obj\ProjectEntries.o
	
ProjectEntries.exe : $(OBJS)
	echo $@
	$(CXX) -o $@ $(OBJS) $(LINKFLAGS)
	
obj\ProjectEntries.o : ProjectEntries.cpp obj\Test.o Framework\Test.h Framework\Render.h | obj
	echo $@
	$(CXX) $(CXXFLAGS) -c $< -o $@
	
obj\Main.o : Framework\Main.cpp Framework\Render.h Framework\Test.h | obj
	echo $@
	$(CXX) $(CXXFLAGS) -c $< -o $@
	
obj\Render.o : Framework\Render.cpp Framework\Render.h | obj
	echo $@
	$(CXX) $(CXXFLAGS) -c $< -o $@
	
obj\Test.o : Framework\Test.cpp Framework\Test.h | obj
	echo $@
	$(CXX) $(CXXFLAGS) -c $< -o $@
obj :
	mkdir obj

clean :
	del obj\ProjectEntries.o
	
cleanall :
	del obj\*.o
	
# $@ is The file name of the target.
# $< is The name of the first prerequisite.
